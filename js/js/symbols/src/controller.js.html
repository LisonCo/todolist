1 (function (window) {
  2   "use strict";
  3 
  4   /**
  5    * Takes a model and view and acts as the controller between them
  6    *
  7    * @constructor
  8    * @param {object} model The model instance
  9    * @param {object} view The view instance
 10    */
 11 
 12   function Controller(model, view) {
 13     var self = this;
 14     self.model = model;
 15     self.view = view;
 16 
 17     self.view.bind("newTodo", function (title) {
 18       self.addItem(title);
 19     });
 20 
 21     self.view.bind("itemEdit", function (item) {
 22       self.editItem(item.id);
 23     });
 24 
 25     self.view.bind("itemEditDone", function (item) {
 26       self.editItemSave(item.id, item.title);
 27     });
 28 
 29     self.view.bind("itemEditCancel", function (item) {
 30       self.editItemCancel(item.id);
 31     });
 32 
 33     self.view.bind("itemRemove", function (item) {
 34       self.removeItem(item.id);
 35     });
 36 
 37     self.view.bind("itemToggle", function (item) {
 38       self.toggleComplete(item.id, item.completed);
 39     });
 40 
 41     self.view.bind("removeCompleted", function () {
 42       self.removeCompletedItems();
 43     });
 44 
 45     self.view.bind("toggleAll", function (status) {
 46       self.toggleAll(status.completed);
 47     });
 48   }
 49 
 50   /**
 51    * Loads and initialises the view
 52    *
 53    * @param {string} '' | 'active' | 'completed'
 54    */
 55   Controller.prototype.setView = function (locationHash) {
 56     var route = locationHash.split("/")[1];
 57     var page = route || "";
 58     this._updateFilterState(page);
 59   };
 60 
 61   /**
 62    * An event to fire on load. Will get all items and display them in the
 63    * todo-list
 64    */
 65   Controller.prototype.showAll = function () {
 66     var self = this;
 67     self.model.read(function (data) {
 68       self.view.render("showEntries", data);
 69     });
 70   };
 71 
 72   /**
 73    * Renders all active tasks
 74    */
 75   Controller.prototype.showActive = function () {
 76     var self = this;
 77     self.model.read({ completed: false }, function (data) {
 78       self.view.render("showEntries", data);
 79     });
 80   };
 81 
 82   /**
 83    * Renders all completed tasks
 84    */
 85   Controller.prototype.showCompleted = function () {
 86     var self = this;
 87     self.model.read({ completed: true }, function (data) {
 88       self.view.render("showEntries", data);
 89     });
 90   };
 91 
 92   /**
 93    * An event to fire whenever you want to add an item. Simply pass in the event
 94    * object and it'll handle the DOM insertion and saving of the new item.
 95    */
 96   Controller.prototype.addItem = function (title) {
 97     var self = this;
 98 
 99     if (title.trim() === "") {
100       return;
101     }
102 
103     self.model.create(title, function () {
104       self.view.render("clearNewTodo");
105       self._filter(true);
106     });
107   };
108 
109   /*
110    * Triggers the item editing mode.
111    */
112   Controller.prototype.editItem = function (id) {
113     var self = this;
114     self.model.read(id, function (data) {
115       self.view.render("editItem", { id: id, title: data[0].title });
116     });
117   };
118 
119   /*
120    * Finishes the item editing mode successfully.
121    */
122   Controller.prototype.editItemSave = function (id, title) {
123     var self = this;
124 
125     while (title[0] === " ") {
126       title = title.slice(1);
127     }
128 
129     while (title[title.length - 1] === " ") {
130       title = title.slice(0, -1);
131     }
132 
133     if (title.length !== 0) {
134       self.model.update(id, { title: title }, function () {
135         self.view.render("editItemDone", { id: id, title: title });
136       });
137     } else {
138       self.removeItem(id);
139     }
140   };
141 
142   /*
143    * Cancels the item editing mode.
144    */
145   Controller.prototype.editItemCancel = function (id) {
146     var self = this;
147     self.model.read(id, function (data) {
148       self.view.render("editItemDone", { id: id, title: data[0].title });
149     });
150   };
151 
152   /**
153    * By giving it an ID it'll find the DOM element matching that ID,
154    * remove it from the DOM and also remove it from storage.
155    *
156    * @param {number} id The ID of the item to remove from the DOM and
157    * storage
158    */
159   Controller.prototype.removeItem = function (id) {
160     var self = this;
161     var items;
162     self.model.read(function (data) {
163       items = data;
164     });
165 
166     self.model.remove(id, function () {
167       self.view.render("removeItem", id);
168     });
169 
170     self._filter();
171   };
172 
173   /**
174    * Will remove all completed items from the DOM and storage.
175    */
176   Controller.prototype.removeCompletedItems = function () {
177     var self = this;
178     self.model.read({ completed: true }, function (data) {
179       data.forEach(function (item) {
180         self.removeItem(item.id);
181       });
182     });
183 
184     self._filter();
185   };
186 
187   /**
188    * Give it an ID of a model and a checkbox and it will update the item
189    * in storage based on the checkbox's state.
190    *
191    * @param {number} id The ID of the element to complete or uncomplete
192    * @param {object} checkbox The checkbox to check the state of complete
193    *                          or not
194    * @param {boolean|undefined} silent Prevent re-filtering the todo items
195    */
196   Controller.prototype.toggleComplete = function (id, completed, silent) {
197     var self = this;
198     self.model.update(id, { completed: completed }, function () {
199       self.view.render("elementComplete", {
200         id: id,
201         completed: completed,
202       });
203     });
204 
205     if (!silent) {
206       self._filter();
207     }
208   };
209 
210   /**
211    * Will toggle ALL checkboxes' on/off state and completeness of models.
212    * Just pass in the event object.
213    */
214   Controller.prototype.toggleAll = function (completed) {
215     var self = this;
216     self.model.read({ completed: !completed }, function (data) {
217       data.forEach(function (item) {
218         self.toggleComplete(item.id, completed, true);
219       });
220     });
221 
222     self._filter();
223   };
224 
225   /**
226    * Updates the pieces of the page which change depending on the remaining
227    * number of todos.
228    */
229   Controller.prototype._updateCount = function () {
230     var self = this;
231     self.model.getCount(function (todos) {
232       self.view.render("updateElementCount", todos.active);
233       self.view.render("clearCompletedButton", {
234         completed: todos.completed,
235         visible: todos.completed > 0,
236       });
237       self.view.render("toggleAll", {
238         checked: todos.completed === todos.total,
239       });
240       self.view.render("contentBlockVisibility", { visible: todos.total > 0 });
241     });
242   };
243 
244   /**
245    * Re-filters the todo items, based on the active route.
246    * @param {boolean|undefined} force  forces a re-painting of todo items.
247    */
248   Controller.prototype._filter = function (force) {
249     var activeRoute =
250       this._activeRoute.charAt(0).toUpperCase() + this._activeRoute.substr(1);
251 
252     // Update the elements on the page, which change with each completed todo
253     this._updateCount();
254 
255     // If the last active route isn't "All", or we're switching routes, we
256     // re-create the todo item elements, calling:
257     //   this.show[All|Active|Completed]();
258     if (
259       force ||
260       this._lastActiveRoute !== "All" ||
261       this._lastActiveRoute !== activeRoute
262     ) {
263       this["show" + activeRoute]();
264     }
265 
266     this._lastActiveRoute = activeRoute;
267   };
268 
269   /**
270    * Simply updates the filter nav's selected states
271    */
272   Controller.prototype._updateFilterState = function (currentPage) {
273     // Store a reference to the active route, allowing us to re-filter todo
274     // items as they are marked complete or incomplete.
275     this._activeRoute = currentPage;
276 
277     if (currentPage === "") {
278       this._activeRoute = "All";
279     }
280 
281     this._filter();
282 
283     this.view.render("setFilter", currentPage);
284   };
285 
286   // Export to window
287   window.app = window.app || {};
288   window.app.Controller = Controller;
289 })(window);
