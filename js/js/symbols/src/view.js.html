1 /*global qs, qsa, $on, $parent, $delegate */
  2 
  3 (function (window) {
  4   "use strict";
  5 
  6   /**
  7    * View that abstracts away the browser's DOM completely.
  8    * It has two simple entry points:
  9    *
 10    *   - bind(eventName, handler)
 11    *     Takes a todo application event and registers the handler
 12    *   - render(command, parameterObject)
 13    *     Renders the given command with the options
 14    */
 15   function View(template) {
 16     this.template = template;
 17 
 18     this.ENTER_KEY = 13;
 19     this.ESCAPE_KEY = 27;
 20 
 21     this.$todoList = qs(".todo-list");
 22     this.$todoItemCounter = qs(".todo-count");
 23     this.$clearCompleted = qs(".clear-completed");
 24     this.$main = qs(".main");
 25     this.$footer = qs(".footer");
 26     this.$toggleAll = qs(".toggle-all");
 27     this.$newTodo = qs(".new-todo");
 28   }
 29 
 30   View.prototype._removeItem = function (id) {
 31     var elem = qs('[data-id="' + id + '"]');
 32 
 33     if (elem) {
 34       this.$todoList.removeChild(elem);
 35     }
 36   };
 37 
 38   View.prototype._clearCompletedButton = function (completedCount, visible) {
 39     this.$clearCompleted.innerHTML =
 40       this.template.clearCompletedButton(completedCount);
 41     this.$clearCompleted.style.display = visible ? "block" : "none";
 42   };
 43 
 44   View.prototype._setFilter = function (currentPage) {
 45     qs(".filters .selected").className = "";
 46     qs('.filters [href="#/' + currentPage + '"]').className = "selected";
 47   };
 48 
 49   View.prototype._elementComplete = function (id, completed) {
 50     var listItem = qs('[data-id="' + id + '"]');
 51 
 52     if (!listItem) {
 53       return;
 54     }
 55 
 56     listItem.className = completed ? "completed" : "";
 57 
 58     // In case it was toggled from an event and not by clicking the checkbox
 59     qs("input", listItem).checked = completed;
 60   };
 61 
 62   View.prototype._editItem = function (id, title) {
 63     var listItem = qs('[data-id="' + id + '"]');
 64 
 65     if (!listItem) {
 66       return;
 67     }
 68 
 69     listItem.className = listItem.className + " editing";
 70 
 71     var input = document.createElement("input");
 72     input.className = "edit";
 73 
 74     listItem.appendChild(input);
 75     input.focus();
 76     input.value = title;
 77   };
 78 
 79   View.prototype._editItemDone = function (id, title) {
 80     var listItem = qs('[data-id="' + id + '"]');
 81 
 82     if (!listItem) {
 83       return;
 84     }
 85 
 86     var input = qs("input.edit", listItem);
 87     listItem.removeChild(input);
 88 
 89     listItem.className = listItem.className.replace("editing", "");
 90 
 91     qsa("label", listItem).forEach(function (label) {
 92       label.textContent = title;
 93     });
 94   };
 95 
 96   View.prototype.render = function (viewCmd, parameter) {
 97     var self = this;
 98     var viewCommands = {
 99       showEntries: function () {
100         self.$todoList.innerHTML = self.template.show(parameter);
101       },
102       removeItem: function () {
103         self._removeItem(parameter);
104       },
105       updateElementCount: function () {
106         self.$todoItemCounter.innerHTML = self.template.itemCounter(parameter);
107       },
108       clearCompletedButton: function () {
109         self._clearCompletedButton(parameter.completed, parameter.visible);
110       },
111       contentBlockVisibility: function () {
112         self.$main.style.display = self.$footer.style.display =
113           parameter.visible ? "block" : "none";
114       },
115       toggleAll: function () {
116         self.$toggleAll.checked = parameter.checked;
117       },
118       setFilter: function () {
119         self._setFilter(parameter);
120       },
121       clearNewTodo: function () {
122         self.$newTodo.value = "";
123       },
124       elementComplete: function () {
125         self._elementComplete(parameter.id, parameter.completed);
126       },
127       editItem: function () {
128         self._editItem(parameter.id, parameter.title);
129       },
130       editItemDone: function () {
131         self._editItemDone(parameter.id, parameter.title);
132       },
133     };
134 
135     viewCommands[viewCmd]();
136   };
137 
138   View.prototype._itemId = function (element) {
139     var li = $parent(element, "li");
140     return parseInt(li.dataset.id, 10);
141   };
142 
143   View.prototype._bindItemEditDone = function (handler) {
144     var self = this;
145     $delegate(self.$todoList, "li .edit", "blur", function () {
146       if (!this.dataset.iscanceled) {
147         handler({
148           id: self._itemId(this),
149           title: this.value,
150         });
151       }
152     });
153 
154     $delegate(self.$todoList, "li .edit", "keypress", function (event) {
155       if (event.keyCode === self.ENTER_KEY) {
156         // Remove the cursor from the input when you hit enter just like if it
157         // were a real form
158         this.blur();
159       }
160     });
161   };
162 
163   View.prototype._bindItemEditCancel = function (handler) {
164     var self = this;
165     $delegate(self.$todoList, "li .edit", "keyup", function (event) {
166       if (event.keyCode === self.ESCAPE_KEY) {
167         this.dataset.iscanceled = true;
168         this.blur();
169 
170         handler({ id: self._itemId(this) });
171       }
172     });
173   };
174 
175   View.prototype.bind = function (event, handler) {
176     var self = this;
177     if (event === "newTodo") {
178       $on(self.$newTodo, "change", function () {
179         handler(self.$newTodo.value);
180       });
181     } else if (event === "removeCompleted") {
182       $on(self.$clearCompleted, "click", function () {
183         handler();
184       });
185     } else if (event === "toggleAll") {
186       $on(self.$toggleAll, "click", function () {
187         handler({ completed: this.checked });
188       });
189     } else if (event === "itemEdit") {
190       $delegate(self.$todoList, "li label", "dblclick", function () {
191         handler({ id: self._itemId(this) });
192       });
193     } else if (event === "itemRemove") {
194       $delegate(self.$todoList, ".destroy", "click", function () {
195         handler({ id: self._itemId(this) });
196       });
197     } else if (event === "itemToggle") {
198       $delegate(self.$todoList, ".toggle", "click", function () {
199         handler({
200           id: self._itemId(this),
201           completed: this.checked,
202         });
203       });
204     } else if (event === "itemEditDone") {
205       self._bindItemEditDone(handler);
206     } else if (event === "itemEditCancel") {
207       self._bindItemEditCancel(handler);
208     }
209   };
210 
211   // Export to window
212   window.app = window.app || {};
213   window.app.View = View;
214 })(window);
