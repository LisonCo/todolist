  2 (function (window) {

  5   /**
  6    * Creates a new client side storage object and will create an empty
  7    * collection if no collection already exists.
  8    *
  9    * @param {string} name The name of our DB we want to use
 10    * @param {function} callback Our fake DB uses callbacks because in
 11    * real life you probably would be making AJAX calls
 12    */
 13   function Store(name, callback) {
 14     callback = callback || function () {};
 15 
 16     this._dbName = name;
 17 
 18     if (!localStorage[name]) {
 19       var data = {
 20         todos: [],
 21       };
 22 
 23       localStorage[name] = JSON.stringify(data);
 24     }
 25 
 26     callback.call(this, JSON.parse(localStorage[name]));
 27   }
 28 
 29   /**
 30    * Finds items based on a query given as a JS object
 31    *
 32    * @param {object} query The query to match against (i.e. {foo: 'bar'})
 33    * @param {function} callback	 The callback to fire when the query has
 34    * completed running
 35    *
 36    * @example
 37    * db.find({foo: 'bar', hello: 'world'}, function (data) {
 38    *	 // data will return any items that have foo: bar and
 39    *	 // hello: world in their properties
 40    * });
 41    */
 42   Store.prototype.find = function (query, callback) {
 43     if (!callback) {
 44       return;
 45     }
 46 
 47     var todos = JSON.parse(localStorage[this._dbName]).todos;
 48 
 49     callback.call(
 50       this,
 51       todos.filter(function (todo) {
 52         for (var q in query) {
 53           if (query[q] !== todo[q]) {
 54             return false;
 55           }
 56         }
 57         return true;
 58       })
 59     );
 60   };
 61 
 62   /**
 63    * Will retrieve all data from the collection
 64    *
 65    * @param {function} callback The callback to fire upon retrieving data
 66    */
 67   Store.prototype.findAll = function (callback) {
 68     callback = callback || function () {};
 69     callback.call(this, JSON.parse(localStorage[this._dbName]).todos);
 70   };
 71 
 72   /**
 73    * Will save the given data to the DB. If no item exists it will create a new
 74    * item, otherwise it'll simply update an existing item's properties
 75    *
 76    * @param {object} updateData The data to save back into the DB
 77    * @param {function} callback The callback to fire after saving
 78    * @param {number} id An optional param to enter an ID of an item to update
 79    */
 80   Store.prototype.save = function (updateData, callback, id) {
 81     var data = JSON.parse(localStorage[this._dbName]);
 82     var todos = data.todos;
 83 
 84     callback = callback || function () {};
 85 
 86     // Generate an ID
 87     var newId = "";
 88     var charset = "0123456789";
 89 
 90     for (var i = 0; i < 6; i++) {
 91       newId += charset.charAt(Math.floor(Math.random() * charset.length));
 92     }
 93 
 94     // If an ID was actually given, find the item and update each property
 95     if (id) {
 96       for (var i = 0; i < todos.length; i++) {
 97         if (todos[i].id === id) {
 98           for (var key in updateData) {
 99             todos[i][key] = updateData[key];
100           }
101           break;
102         }
103       }
104 
105       localStorage[this._dbName] = JSON.stringify(data);
106       callback.call(this, todos);
107     } else {
108       // Assign an ID
109       updateData.id = parseInt(newId);
110 
111       todos.push(updateData);
112       localStorage[this._dbName] = JSON.stringify(data);
113       callback.call(this, [updateData]);
114     }
115   };
116 
117   /**
118    * Will remove an item from the Store based on its ID
119    *
120    * @param {number} id The ID of the item you want to remove
121    * @param {function} callback The callback to fire after saving
122    */
123   Store.prototype.remove = function (id, callback) {
124     var data = JSON.parse(localStorage[this._dbName]);
125     var todos = data.todos;
126     // var todoId;
127 
128     for (var i = 0; i < todos.length; i++) {
129       if (todos[i].id === id) {
130         todos.splice(i, 1);
131       }
132     }
133 
134     // for (var i = 0; i < todos.length; i++) {
135     //   if (todos[i].id == todoId) {
136     //     todos.splice(i, 1);
137     //   }
138     // }
139 
140     localStorage[this._dbName] = JSON.stringify(data);
141     callback.call(this, todos);
142   };
143 
144   /**
145    * Will drop all storage and start fresh
146    *
147    * @param {function} callback The callback to fire after dropping the data
148    */
149   Store.prototype.drop = function (callback) {
150     var data = { todos: [] };
151     localStorage[this._dbName] = JSON.stringify(data);
152     callback.call(this, data.todos);
153   };
